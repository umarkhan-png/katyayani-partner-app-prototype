09 · Motion & Animation
Token Reference
/* Micro-interactions */
transition: all .15s ease           — pack tile hover, add-btn, chip select
transition: all .2s ease            — chevron rotate, header appear
transition: background 1s ease      — hero gradient (festive mode toggle)
transition: max-height .4s ease     — sticky category bar slide down

/* Entrance animations */
@keyframes grow        — bar fill (stock/progress bar, 1s ease forwards, scaleX from 0)
@keyframes pulse-ring  — notification ring (1.8s ease-out infinite, scale .7→2.2)
@keyframes aiFabPulse  — AI FAB glow (2.8s ease-in-out infinite, box-shadow breathe)

/* Celebration */
@keyframes confetti-fall  — order success confetti (2–2.2s ease-out, translate + rotate)
@keyframes celeb-shield   — lock unlock shield pop (2.6s cubic-bezier .34 1.56 .64 1)
@keyframes celeb-text     — success text enter (2.4s ease-out, translateY 16px→0)

/* Festive */
@keyframes sparkle  — gold star twinkle (2.5s ease-in-out infinite)
@keyframes float    — diya float (3s ease-in-out infinite, translateY 0 → -6px)
@keyframes swing    — garland sway (4s ease-in-out infinite, rotate ±2deg)

/* Rule: all interactive elements need a .15s transition for touch feedback */
